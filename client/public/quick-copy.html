2. Digital Ocean Commands

# Digital Ocean Deployment Commands
ssh root@YOUR_SERVER_IP

# Install Node.js
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
apt-get install -y nodejs

# Create app directory
mkdir -p /var/www/pulseboardai
cd /var/www/pulseboardai

# Upload your files here, then:
npm install
npm run build
npm install -g pm2
pm2 start dist/index.js --name pulseboardai
pm2 save
pm2 startup
